CREATE TABLE MEMO_KKW (	
	MEMO_NUM NUMBER,
	USER_ID VARCHAR2(200 BYTE) DEFAULT NULL, 
	USER_NAME VARCHAR2(200 BYTE), 
	TITLE VARCHAR2(200 BYTE) DEFAULT NULL,
    CONTENT CLOB, 
    CONTENT_REPLY CLOB, 
    NOTICE_YN VARCHAR2(1 BYTE) DEFAULT 'N', 
    WRITE_DATE DATE DEFAULT null, 
	REGIST_DATE DATE DEFAULT null, 
	UPDATE_DATE DATE DEFAULT null, 
	REGIST_ID VARCHAR2(200 BYTE) DEFAULT NULL, 
	UPDATE_ID VARCHAR2(200 BYTE) DEFAULT NULL, 
	REGIST_IP VARCHAR2(200 BYTE) DEFAULT NULL, 
	UPDATE_IP VARCHAR2(200 BYTE) DEFAULT NULL, 
    PRIMARY KEY (MEMO_NUM)
);
